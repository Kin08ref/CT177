int member(int x, List L)
{
    for(int i=0; i<L.Last; i++)
    {
        if(L.Elements[i]==x)
            return 1;
    }
    return 0;
}
void readList(List *L)
{
    makenullList(L);
    int n;
    scanf("%d",&n);
    for (int i=0; i<n; i++)
    {
        scanf("%d",&L->Elements[i]);
        if(!member(L->Elements[i],*L))
            insertList(L->Elements[i],L->Last+1,L);
    }
}
void Even(List L1, List L2)
{
    for(int i=0; i<L1.Last; i++)
        {printf("%d ",L1.Elements[i]);}
    for(int i=0; i<L2.Last;i++)
    {
        int count=0;
        for(int z=0; z<L1.Last; z++)
        {
            if(L2.Elements[i]==L1.Elements[z])
               count++;
        } 
        if(count==0)
            printf("%d ",L2.Elements[i]);
    }
}
void printList(List L)
{
    for (int i=0; i<L.Last; i++)
    {
        printf("%d ", L.Elements[i]);
    }
    printf("\n");
}
